JavaScript 基本構文

- if 文 -

                    
    if (条件式) {
        trueブロック
    } else if (条件式) {
    　次のtrueブロック
    } else {
        falseブロック
    }

    //例
    let x = 30;

    if (x >= 20) {
        console.log('変数xは20以上です。');
    } else if (x >= 10) {
        console.log('変数xは10以上です。');
    } else {
        console.log('変数xは10未満です。');
    }

- switch 文 -

                    
    switch (式) {
        case 値1:
        「式 === 値1」である場合の命令
            break;
        case 値2:
        「式 === 値2」である場合の命令
            break;
        default:
            式の値がすべての値に合致しない場合の命令
            break;
    }
    
    //例
    let rank = 'B';
    switch(rank) {
        case 'A' :
            console.log('Aランクです。');
            break;
        case 'B' :
            console.log('Bランクです。');
            break;
        case 'C' :
            console.log('Cランクです。');
            break;
        default : 
            console.log('ランク外です。');
            break;
    }
    
    //フォールスルーさせる場合
    switch(rank) {
        case 'A' :
        case 'B' :
        case 'C' :
            console.log('合格です！');
            break;
        default : 
            console.log('不合格です！');
            break;
    }

- for 文 -

                    
    for (初期化式; ループ継続条件式; 増減式) {
        ループ内で実行する命令
    }
    
    //配列を取り出すときは、
    for (let i = 0; i < array.length; i++) {
        console.log(array[i]);
    }

- while 文 -

                    
    while (条件式) {
        条件がtrueのときの命令
    }
    
    //例
    let i = 5;
    while(i < 10) {
        console.log(`iの値は${i}`);
        i++;
    }

- for in：オブジェクトの中身を取り出す -

                    
    for (仮変数 in オブジェクト名) {
        ループ内で実行する命令
    }
    
    //例
    let data = {
        apple: 150,
        orange: 100,
        banana: 120
    };
    
    for (let key in data) {
        console.log(`${key} = ${data[key]}`);
    }
    
    //結果
    apple = 150
    orange = 100
    banana = 120

- for of：配列の中身を取り出す -

                    
    for (仮変数 of 配列名) {
        ループ内で実行する命令
    }
    
    //例
    let data = ['apple', 'orange', 'banana'];
    for (let value of data) {
        console.log(value);
    }
    
    //結果
    apple
    orange
    banana

- break（一番内側のループを脱出） -

                    
    let k = 1;
    let result = 0;
    
    for (k = 1; k <= 100; k++) {
        result += k;
        if (result > 1000) { break; }
    }
    
    console.log(`合計値が1000を超えるのは${k}`);

- continue （特定の条件下の周回をスキップ） -

                    
    let result = 0;

    for (let m = 1; m < 100; m++) {
        if (m % 2 === 0) { continue; }
        result += m;
        console.log(`${result-m} + ${m} = ${result}`);
    }
    
    console.log(`合計：${result}`);
